    def _make_func(self, names):
        from types import MethodType

        def make_func(name):
            def func(self, *args):
                args_with_name = [
                    name,
                ] + list(args)
                return self.add(*args_with_name)

            func.__doc__ = SHAPE_EXAMPLES[name]
            return func

        for name in names:
            func_name = 'add_' + name
            func = make_func(name)
            setattr(self, func_name, MethodType(func, self))
